MORE ABOUT STATE:

->Re-rendering works only if the state value is changed.

let [count,setCount]=useState(0);
// In this useSate(0) we can even pass the function.


 import { useState } from "react";
   function init()
    {
        console.log("init was called")
        return Math.random();
    }
 function Counter()
{
  

    let [count,setCount]=useState(init); --> here we need pass the reference of the function not the function call (init()). if we directly call the function what ever written in the funtion will again get executed.

    
    console.log("The counter is getting re-rendered")
    console.log("count ",count)
    
   let incCount=()=>{
        setCount((currCount)=>{
        return currCount+1
    })

    
   }

   return (
    <div>
        <h3> Count = {count}</h3>
        <button onClick={incCount}>Increase count</button>
    </div>
   )
}
export default Counter;
